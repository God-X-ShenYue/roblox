--Value
_G.ShenYuefa = true
_G.ShenYuemode = Value
_G.ShenYuestring =Value

--function
function ShenYuefa()
   while _G.ShenYuefa == true do
    game:GetService("ReplicatedStorage").ChatEvents.SendMessage:InvokeServer({["Message"] = "".._G.ShenYuestring ,["Channel"] = "Server",["ChatType"] = "Match",["lastPlayerIsTheSame"] = _G.ShenYuemode,["messageChannel"] = game:GetService("Players").LocalPlayer.PlayerGui.EchoChat.UI.Channels.Server})
    wait(_G.ShenYuespeed)
   end
end
